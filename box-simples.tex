  [leftbartext]
  [
    width=broad,
    frame=off,
    framecolor=darkgray,
    leftframe=on,
    rulethickness=2ex,
    offset=0.25ex,
    loffset=3ex, 
    background=color,
    backgroundcolor=lightgray,
  ]

\definecolor[darkred]   [r=0.8]
\definecolor[darkgreen] [g=0.8]
\definecolor[darkblue]  [b=0.8]
